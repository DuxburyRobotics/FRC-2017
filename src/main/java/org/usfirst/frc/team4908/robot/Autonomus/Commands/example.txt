#The Basic Idea _command_,enable/dissable OR speed,_time till next command_



#To drive straight at a speed of 10 for 3 seconds
Drive,10,3

#To enable the Intake for 3 seconds
Intake,enable,3

#dissable the intake for 3 seconds
Intake,disable,3

#to enable the Shooter for 3 seconds
Shoot,enable,3

#To disable the intake for 3 seconds
Shoot,disable,3
